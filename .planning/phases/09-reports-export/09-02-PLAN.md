         async function handleDownload(type: 'pdf' | 'csv' | 'xlsx') {
           setDownloading(type)
           try {
             const params = type === 'pdf'
               ? `/api/reports/pdf?runId=${runId}`
               : `/api/reports/export?datasetId=${datasetId}&format=${type}&runId=${runId}`
             const res = await fetch(params)
             if (!res.ok) throw new Error('Download failed')
             const blob = await res.blob()
             const url = URL.createObjectURL(blob)
             const a = document.createElement('a')
             a.href = url
             a.download = type === 'pdf' ? `qc-report.pdf` : `dataset-annotated.${type}`
             a.click()
             URL.revokeObjectURL(url)
           } catch { /* show error via console.error */ }
           finally { setDownloading(null) }
         }
